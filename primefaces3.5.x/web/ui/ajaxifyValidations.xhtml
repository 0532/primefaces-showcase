AJAX - Validation

Validations are executed on the server side and page is updated with the response. In this example, ajax validations are executed when input field loses focus(blur).

			<p:inputText id="firstname" 
					value="\#{pprBean.firstname}" label="Firstname">
				<f:validateLength minimum="5" />
				<p:ajax event="blur" update="msg" />
			</p:inputText>

package org.primefaces.examples.view;

import javax.faces.application.FacesMessage;
import javax.faces.context.FacesContext;
import javax.faces.event.ActionEvent;

public class PersonBean {

	private String firstname;
	
	private String surname;

	public String getFirstname() {
		return firstname;
	}
	public void setFirstname(String firstname) {
		this.firstname = firstname;
	}

	public String getSurname() {
		return surname;
	}
	public void setSurname(String surname) {
		this.surname = surname;
	}
	
	public void savePerson(ActionEvent actionEvent) {
		FacesContext.getCurrentInstance().addMessage(null, new FacesMessage("Welcome " + firstname + " " + surname + "!"));
	}
}
